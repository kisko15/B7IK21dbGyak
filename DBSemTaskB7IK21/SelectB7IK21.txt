	SELECT Futocipo_neve FROM Futocipo WHERE Teljes_ar > ANY (SELECT Teljes_ar FROM Futocipo F LEFT OUTER JOIN Meret M ON F.FutocipoID=M.Meret WHERE Meret=42);
	π Futocipo_neve σ Teljes_ar > ( π Teljes_ar σ Meret=42 Futocipo) Futocipo
